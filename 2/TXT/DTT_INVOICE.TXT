CREATE OR REPLACE FUNCTION FN_TRG_HST_INVOICE()
 RETURNS trigger AS 
 $BODY$ 
 DECLARE
 vSEQ BIGINT; 
 BEGIN 
 SELECT COALESCE(MAX(HST_ID),0)+1 INTO vSEQ FROM HST_INVOICE ;
 IF TG_OP = 'INSERT' THEN 
 INSERT INTO HST_INVOICE SELECT vSEQ , NEW.* ; 
 ELSIF TG_OP = 'UPDATE' THEN 
 INSERT INTO HST_INVOICE SELECT vSEQ , NEW.* ; 

 END IF; 
 RETURN NEW; 
 END; 
 $BODY$ 
 LANGUAGE plpgsql ; 

 CREATE TRIGGER TRG_HST_INVOICE
 AFTER INSERT Or UPDATE 
 ON INVOICE
 FOR EACH ROW 
 EXECUTE PROCEDURE FN_TRG_HST_INVOICE() ;


ALTER TABLE INVOICE  DISABLE TRIGGER TRG_HST_INVOICE ;
